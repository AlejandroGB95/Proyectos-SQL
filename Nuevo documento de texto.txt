DECLARE
	v_num_empleados NUMBER(2);
BEGIN
	INSERT INTO DEPART VALUES (99,'PROVISIONAL',NULL);
	UPDATE EMPLE SET DEPT_NO = 99 WHERE DEPT_NO=20;
	v_num_empleados := SQL%ROWCOUNT;
	DELETE FROM DEPART WHERE DEPT_NO=20;
	DBMS_OUTPUT.PUT_LINE (v_num_empleados || ' Empleados ubicados en PROVISIONAL');
EXCEPTION
	WHEN OTHERS THEN 
	RAISE_APPLICATION_ERROR(-20000, 'Error en aplicación');
END;
/

DECLARE
	v_cadena VARCHAR2(10);
	contador INTEGER;
BEGIN
	contador := LENGTH('HOLA');
	LOOP
		v_cadena := v_cadena||SUBSTR('HOLA',contador,1);
		contador := contador - 1;
	EXIT WHEN contador=0;
	END LOOP;
	DBMS_OUTPUT.PUT_LINE(v_cadena);
END;

CREATE OR REPLACE PROCEDURE cambiar_depart (num_empleado NUMBER,nuevo_depart NUMBER)
AS
	v_depart_antiguo EMPLE.DEPT_NO%TYPE;
BEGIN 
	select dept_no into v_depart_antiguo FROM emple where emp_no = num_empleado;
	UPDATE emple SET DEPT_NO = nuevo_depart where emp_no=num_empleado;
	DBMS_OUTPUT.PUT_LINE(num_empleado||' Departamento anterior: '||v_depart_antiguo||'Departamento nuevo: '||nuevo_depart);
EXCEPTION
	WHEN NO_DATA_FOUND THEN 
		RAISE_APPLICATION_ERROR (-2000,'Error no hay datos');
	WHEN OTHERS THEN
		RAISE_APPLICATION_ERROR (-2000,'Error en la aplicación');
END;
/	

	
CREATE OR REPLACE PROCEDURE suma_dos (num_1 NUMBER, num_2 NUMBER)
AS
	v_suma NUMBER;
BEGIN
	v_suma := num_1+num_2;
	DBMS_OUTPUT.PUT_LINE('La suma de '||num_1||' y '||num_2||' es: '||v_suma);
END;
/
	
CREATE OR REPLACE PROCEDURE reverso_cadena (cadena_original VARCHAR2)
AS	
	vr_cadena VARCHAR2;
BEGIN
	FOR i IN REVERSE 1..LENGTH(cadena_original) LOOP
	vr_cadena := vr_cadena|| substr(cadena_original,i,1);
	END LOOP;
	DBMS_OUTPUT.PUT_LINE(vr_cadena);
END;
/
